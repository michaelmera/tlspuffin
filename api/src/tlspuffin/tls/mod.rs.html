/// Function symbol which can be used for debugging
///
#[allow(dead_code)]
fn fn_debug(
    message: &rustls::msgs::message::Message,
) -> Result<rustls::msgs::message::Message, crate::tls::error::FnError> {
    dbg!(message);
    Ok(message.clone())
}
